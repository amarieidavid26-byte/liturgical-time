static async getOrCreateOrthodoxCalendar(): Promise<string> {
  const calendars = await Calendar.getCalendarsAsync(Calendar.EntityTypes.EVENT);
  
  // Platform-specific source selection
  let calendarSource;
  if (Platform.OS === 'ios') {
    calendarSource = calendars.find(cal => cal.source.name === 'iCloud')?.source;
  } else {
    // Android - prefer Google account, fallback to local
    calendarSource = calendars.find(cal => cal.source.name === 'Google')?.source ||
                    { isLocalAccount: true, name: 'Timpul Liturgic', type: 'LOCAL' };
  }
  
  // Create calendar with platform-appropriate settings
  const calendarId = await Calendar.createCalendarAsync({
    title: 'Timpul Liturgic',
    color: Platform.OS === 'ios' ? '#800020' : '#800020FF', // Android needs alpha
    entityType: Calendar.EntityTypes.EVENT,
    sourceId: calendarSource?.id,
    source: calendarSource,
    name: 'ortodox',
    ownerAccount: Platform.OS === 'android' ? 'liturgical.time@local' : 'personal',
    accessLevel: Calendar.CalendarAccessLevel.OWNER,
  });
  
  return calendarId;
}"
