"Create a new Prayer Logger tab with customizable reminder notifications:

1. CREATE new tab file app/(tabs)/prayers.tsx:

import React, { useState, useEffect } from 'react';
import { View, Text, ScrollView, TouchableOpacity, TextInput, Switch, Alert } from 'react-native';
import DateTimePicker from '@react-native-community/datetimepicker';
import * as Notifications from 'expo-notifications';
import { Ionicons } from '@expo/vector-icons';
import useAppStore from '../../lib/store/appStore';
import Colors from '../../constants/Colors';

export default function PrayersScreen() {
  const [prayers, setPrayers] = useState([]);
  const [showAddModal, setShowAddModal] = useState(false);
  
  // Prayer form states
  const [prayerName, setPrayerName] = useState('');
  const [prayerTime, setPrayerTime] = useState(new Date());
  const [reminderMinutes, setReminderMinutes] = useState(30);
  const [isDaily, setIsDaily] = useState(true);
  const [selectedDays, setSelectedDays] = useState([1,2,3,4,5,6,0]); // All days
  
  const requestNotificationPermissions = async () => {
    const { status } = await Notifications.requestPermissionsAsync();
    return status === 'granted';
  };
  
  const schedulePrayerNotification = async (prayer) => {
    const hasPermission = await requestNotificationPermissions();
    if (!hasPermission) return null;
    
    // Calculate notification time (prayer time minus reminder minutes)
    const notificationTime = new Date(prayer.time);
    notificationTime.setMinutes(notificationTime.getMinutes() - prayer.reminderMinutes);
    
    const notificationId = await Notifications.scheduleNotificationAsync({
      content: {
        title: '🕊️ Timp de Rugăciune',
        body: `${prayer.name} în ${prayer.reminderMinutes} minute`,
        sound: 'default',
        data: { prayerId: prayer.id },
      },
      trigger: {
        hour: notificationTime.getHours(),
        minute: notificationTime.getMinutes(),
        repeats: prayer.isDaily,
        weekday: prayer.isDaily ? undefined : prayer.selectedDays,
      },
    });
    
    return notificationId;
  };
  
  const addPrayer = async () => {
    const newPrayer = {
      id: Date.now(),
      name: prayerName,
      time: prayerTime,
      reminderMinutes,
      isDaily,
      selectedDays,
      notificationId: null,
    };
    
    // Schedule notification
    const notificationId = await schedulePrayerNotification(newPrayer);
    newPrayer.notificationId = notificationId;
    
    // Save to database
    await savePrayer(newPrayer);
    setPrayers([...prayers, newPrayer]);
    setShowAddModal(false);
  };
  
  return (
    <ScrollView style={styles.container}>
      {/* Header with gradient */}
      <LinearGradient colors={Colors.orthodox.royalBlueGradient} style={styles.header}>
        <Text style={styles.headerTitle}>Jurnal de Rugăciune</Text>
        <Text style={styles.headerSubtitle}>Programează-ți timpul cu Domnul</Text>
      </LinearGradient>
      
      {/* Prayer List */}
      {prayers.map(prayer => (
        <View key={prayer.id} style={styles.prayerCard}>
          <View style={styles.prayerHeader}>
            <Text style={styles.prayerName}>{prayer.name}</Text>
            <Text style={styles.prayerTime}>
              {format(prayer.time, 'HH:mm')}
            </Text>
          </View>
          <View style={styles.reminderBadge}>
            <Ionicons name='notifications' size={16} color={Colors.orthodox.royalBlue} />
            <Text style={styles.reminderText}>
              Amintire cu {prayer.reminderMinutes} min înainte
            </Text>
          </View>
        </View>
      ))}
      
      {/* Add Prayer Button */}
      <TouchableOpacity style={styles.addButton} onPress={() => setShowAddModal(true)}>
        <Ionicons name='add-circle' size={24} color='white' />
        <Text style={styles.addButtonText}>Adaugă Rugăciune</Text>
      </TouchableOpacity>
      
      {/* Quick Prayer Presets */}
      <View style={styles.presetsSection}>
        <Text style={styles.sectionTitle}>Rugăciuni Rapide</Text>
        
        <TouchableOpacity style={styles.presetButton} onPress={() => {
          setPrayerName('Rugăciunile Dimineții');
          setPrayerTime(new Date(2024, 0, 1, 7, 0));
          setReminderMinutes(15);
          setShowAddModal(true);
        }}>
          <Text>🌅 Rugăciunile Dimineții - 7:00</Text>
        </TouchableOpacity>
        
        <TouchableOpacity style={styles.presetButton} onPress={() => {
          setPrayerName('Rugăciunile Serii');
          setPrayerTime(new Date(2024, 0, 1, 21, 0));
          setReminderMinutes(15);
          setShowAddModal(true);
        }}>
          <Text>🌙 Rugăciunile Serii - 21:00</Text>
        </TouchableOpacity>
        
        <TouchableOpacity style={styles.presetButton} onPress={() => {
          setPrayerName('Paraclisul Maicii Domnului');
          setPrayerTime(new Date(2024, 0, 1, 19, 0));
          setReminderMinutes(30);
          setShowAddModal(true);
        }}>
          <Text>🕊️ Paraclisul Maicii Domnului</Text>
        </TouchableOpacity>
      </View>
    </ScrollView>
  );
}

2. UPDATE lib/database/sqlite.ts to add prayers table:

CREATE TABLE IF NOT EXISTS prayers (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT NOT NULL,
  time TEXT NOT NULL,
  reminderMinutes INTEGER DEFAULT 30,
  isDaily INTEGER DEFAULT 1,
  selectedDays TEXT,
  notificationId TEXT,
  lastCompleted TEXT,
  streak INTEGER DEFAULT 0
);
