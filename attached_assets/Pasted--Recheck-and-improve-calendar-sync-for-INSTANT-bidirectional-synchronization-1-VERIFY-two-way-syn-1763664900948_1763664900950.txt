export class CalendarSyncService {
  static syncInterval: any = null;
  
  // Enable real-time sync
  static enableRealTimeSync(callback: () => void) {
    // Sync every 30 seconds when app is active
    this.syncInterval = setInterval(async () => {
      console.log('⚡ Real-time sync check...');
      
      // Import new events from device calendar
      const imported = await this.smartImportMeetings();
      
      // Check for external changes
      const updated = await this.syncExternalChanges();
      
      if (imported > 0 || updated > 0) {
        console.log(`✅ Synced: ${imported} new, ${updated} updated`);
        callback(); // Refresh UI
      }
    }, 30000); // Every 30 seconds
  }
  
  // Instant push to calendar when meeting created/updated
  static async instantPushToCalendar(meeting: Meeting) {
    try {
      // Delete old event if exists
      if (meeting.calendarEventId) {
        await Calendar.deleteEventAsync(meeting.calendarEventId);
      }
      
      // Create new event immediately
      const eventId = await this.syncMeetingToCalendar(meeting);
      console.log('⚡ Instant push complete:', eventId);
      return eventId;
    } catch (error) {
      console.error('Instant push failed:', error);
    }
  }
  
  // Instant pull from calendar on app focus
  static async instantPullFromCalendar() {
    console.log('⚡ Instant pull starting...');
    const changes = await this.smartImportMeetings();
    console.log('⚡ Instant pull complete:', changes);
    return changes;
  }
}
